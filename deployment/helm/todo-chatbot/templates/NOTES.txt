DEPLOYMENT STATUS:

  $ kubectl get pods -l app.kubernetes.io/name={{ include "todo-chatbot.name" . }}

Check deployment status:
  $ kubectl get deployments -l app.kubernetes.io/name={{ include "todo-chatbot.name" . }}

Check service status:
  $ kubectl get services -l app.kubernetes.io/name={{ include "todo-chatbot.name" . }}

View logs:
  Frontend: $ kubectl logs -l app.kubernetes.io/component=frontend
  Backend:  $ kubectl logs -l app.kubernetes.io/component=backend

SCALING:

Scale backend replicas:
  $ helm upgrade {{ .Release.Name }} {{ .Chart.Name }} --set backend.replicaCount=4

Scale frontend replicas:
  $ helm upgrade {{ .Release.Name }} {{ .Chart.Name }} --set frontend.replicaCount=2

TROUBLESHOOTING:

If pods are not starting:
  $ kubectl describe pod <pod-name>
  $ kubectl logs <pod-name>

If services are not accessible:
  $ kubectl get endpoints
  $ kubectl get events --sort-by='.lastTimestamp'
